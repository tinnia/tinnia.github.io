```python
def solution(words):
    answer = 0
    for word in words:
        i = 1
        while 1:
            if i == len(word):
                answer += i
                break
            cnt = 0
            for w in words:
                if word[:i] == w[:i]:
                    cnt += 1
                    if cnt > 1:
                        break
            if cnt == 1:
                answer += i
                break
            i += 1
    return answer
```
