```python
def solution(priorities, location):
    answer = 0
    while 1:
        m = max(priorities)
        tmp = priorities.pop(0)
        if m == tmp:
            answer += 1
            if location == 0:
                break
            else:
                location -= 1
        else:
            priorities.append(tmp)
            if location == 0:
                location = len(priorities) - 1
            else:
                location -= 1
    return answer
```
