```python
def solution(nums):
    from itertools import combinations
    answer = 0

    # ì†Œìˆ˜ dp ë§Œë“¤ê¸°
    dp = [1] * (sum(nums) + 1)
    dp[0] = dp[1] = 0
    i = 2
    while i <= sum(nums):
        for j in range(2 * i, sum(nums) + 1, i):
            dp[j] = 0
        i += 1

    tmp = list(combinations(nums, 3))
    for k in tmp:
        if dp[sum(k)]:
            answer += 1

    return answer
```
