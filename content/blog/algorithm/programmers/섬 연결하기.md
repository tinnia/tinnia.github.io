def solution(n, costs):
    answer = 1e9
    arr = [[0] * n for _ in range(n)]
    for cost in costs:
        arr[cost[0]][cost[1]] = cost[2]
        arr[cost[1]][cost[0]] = cost[2]

    def func(idx, S):
        nonlocal answer
        visit[idx] = 1
        if 0 not in visit:
            answer = min(answer, S)
            return
        for i in range(n):
            if visit[i]: continue
            if arr[idx][i]:
                visit[i] = 1
                func(i, S + arr[idx][i])
                visit[i] = 0

    for i in range(n):
        visit = [0] * n
        func(i, 0)

    return answer
