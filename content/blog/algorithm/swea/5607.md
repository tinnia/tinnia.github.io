def nCr(n, r, mod):
    lst = [1] * (n+1)
    for i in range(1, n+1):
        lst[i] = lst[i-1] * i % mod
    A = lst[n]
    B = pow(lst[r], (mod-2), mod)
    C = pow(lst[n-r], (mod-2), mod)
    return (A * B * C) % mod


for T in range(1, int(input()) + 1):
