def back(k):
    if k==3:
        res.append(Sum(lst,k))
    else:
        for i in range(7):
            if visit[i] : continue
            lst.append(i)
            visit[i] = 1
            back(k+1)
            lst.pop()
            for j in range(i+1,7):
                visit[j] = 0


def Sum(lst,k):
    s=0
    for i in range(k):
        s+=num[lst[i]]
    return s

for tc in range(1, int(input()) + 1):
    num = list(map(int, input().split()))
    visit=[0]*7
    lst=[]
    res=[]
    back(0)
    res=list(set(res))
    res.sort()
    print('#{} {}'.format(tc,res[-5]))
